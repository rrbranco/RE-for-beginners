\subsection{Klassen}

Einfache Klassen:

\begin{lstlisting}[caption=test.java,style=customjava]
public class test
{
        public static int a;
        private static int b;

        public test()
        {
            a=0;
            b=0;
        }
        public static void set_a (int input)
	{
		a=input;
	}
	public static int get_a ()
	{
		return a;
	}
	public static void set_b (int input)
	{
		b=input;
	}
	public static int get_b ()
	{
		return b;
	}
}
